import speech_recognition as sr
recognizer = sr.Recognizer()
with sr.AudioFile("D:\\folder\\myaudio.wav") as source:
audio_data = recognizer.record(source)
try:
text = recognizer.recognize_google(audio_data, language="en-US")
print("Text from audio (English):", text)
except sr.UnknownValueError:
print("Google Web API could not understand the audio (English)")
except sr.RequestError as e:
print(f"Could not request results from Google Web API (English); {e}")
