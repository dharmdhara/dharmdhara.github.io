ऑडियो रिकार्ड का स्वचालित टङ्कण (टाइपिंग)
कृत्रिम बुद्धि के Google Web API द्वारा मुफ्त में अंग्रेजी एवं हिन्दी ऑडियो रिकार्ड के स्वचालित टङ्कण टाइपिंग के पायथन कोड प्रस्तुत हैं । निम्न दोनों कोड को क्रमशः type_eng.py तथा type_hin.py नाम से सुरक्षित कर लें और तब विण्डो में Anaconda को इन्सटॉल करके तब अनाकोण्डा में pip install SpeechRecognition pydub इन्सटॉल करने के पश्चात निम्न स्क्रिप्ट कार्य करेगा । तब python D:\typing\type_eng.py अथवा python D:\typing\type_hin.py रन करने पर निम्न स्क्रिप्ट कार्य करेगा । यदि धीमी गति से स्पष्ट बोलकर रिकार्ड किये हैं तो त्रुटि नहीं अथवा नगण्य होगी ।
type_eng.py =
════════════════════════════════════
import speech_recognition as sr
recognizer = sr.Recognizer()
with sr.AudioFile("D:\\folder\\myaudio.wav") as source:
audio_data = recognizer.record(source)
try:
text = recognizer.recognize_google(audio_data, language="en-US")
print("Text from audio (English):", text)
except sr.UnknownValueError:
print("Google Web API could not understand the audio (English)")
except sr.RequestError as e:
print(f"Could not request results from Google Web API (English); {e}")
