function Get-CPUTemperature {
while ($true) {
$query = Get-WmiObject -Namespace "root/wmi" -Query "SELECT * FROM MSAcpi_ThermalZoneTemperature"
$temperatureKelvin = $query.CurrentTemperature / 10
$temperatureCelsius = [math]::Round(($temperatureKelvin - 273.15), 2)
Write-Host "CPU Temperature: $temperatureCelsius C"
Start-Sleep -Seconds 5
}
}
