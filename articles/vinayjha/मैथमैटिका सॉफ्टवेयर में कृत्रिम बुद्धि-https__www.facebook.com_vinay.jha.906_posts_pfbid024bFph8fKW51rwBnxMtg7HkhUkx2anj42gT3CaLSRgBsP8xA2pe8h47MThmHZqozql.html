{conv, convFinal, leakyReLU, cat},
conv = ConvolutionLayer[32, {3, 3}, "PaddingSize" -> 1];
convFinal = ConvolutionLayer[64, {3, 3}, "PaddingSize" -> 1];
leakyReLU = ElementwiseLayer[Function[Max[0, #] + 0.2*Min[0, #]]];
cat = CatenateLayer[1];
NetGraph[
<|
"conv1" -> conv, "conv2" -> conv, "conv3" -> conv,
"conv4" -> conv, "conv5" -> convFinal,
"lReLU1" -> leakyReLU, "lReLU2" -> leakyReLU,
"lReLU3" -> leakyReLU, "lReLU4" -> leakyReLU,
"cat1" -> cat, "cat2" -> cat, "cat3" -> cat, "cat4" -> cat,
